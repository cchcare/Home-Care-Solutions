function buildSidebar() {
  const nav = document.getElementById('sidebarNav');
  const sections = [
    { label: 'New Hire Orientation', types: ['new-hire'] },
    { label: 'Personal Care Skills', types: ['personal-care'] },
    { label: 'Annual Training', types: ['annual'] }
  ];
  sections.forEach(sec => {
    const mods = modules.filter(m => sec.types.includes(m.type));
    if (!mods.length) return;
    const header = document.createElement('div');
    header.style.cssText = 'font-size:10px;font-weight:700;text-transform:uppercase;letter-spacing:1.5px;color:var(--gray-500);padding:16px 4px 6px;';
    header.textContent = sec.label;
    nav.appendChild(header);
    mods.forEach((mod, idx) => {
      const globalIdx = modules.indexOf(mod);
      const btn = document.createElement('div');
      btn.className = 'sidebar-module';
      btn.innerHTML = `<button class="sidebar-module-btn" id="sidebarBtn_${globalIdx}" onclick="showModule(${globalIdx})">
        <span class="module-icon">${mod.icon}</span>
        <span class="module-label">${mod.title}</span>
        <span class="module-status" id="sidebarStatus_${globalIdx}">○</span>
      </button>`;
      nav.appendChild(btn);
    });
  });
}

function showModule(idx) {
  document.querySelectorAll('.module-card').forEach(c => c.classList.remove('active'));
  document.querySelectorAll('.sidebar-module-btn').forEach(b => b.classList.remove('active'));
  const card = document.getElementById(`module_${idx}`);
  if (card) { card.classList.add('active'); card.scrollIntoView({ behavior: 'smooth', block: 'start' }); }
  const btn = document.getElementById(`sidebarBtn_${idx}`);
  if (btn) btn.classList.add('active');
  currentModule = idx;
  updateProgress();
}

function submitQuiz(modIdx) {
  const mod = modules[modIdx];
  let correct = 0;
  let allAnswered = true;

  mod.questions.forEach((q, qi) => {
    const selected = document.querySelector(`input[name="q_${modIdx}_${qi}"]:checked`);
    if (!selected) { allAnswered = false; return; }
    const selVal = parseInt(selected.value);
    const fb = document.getElementById(`feedback_${modIdx}_${qi}`);

    // Disable all options
    document.querySelectorAll(`input[name="q_${modIdx}_${qi}"]`).forEach(r => r.disabled = true);
    q.options.forEach((opt, oi) => {
      const lbl = document.getElementById(`opt_${modIdx}_${qi}_${oi}`);
      if (oi === q.correct) lbl.classList.add('correct');
      else if (oi === selVal && selVal !== q.correct) lbl.classList.add('incorrect');
    });

    if (selVal === q.correct) {
      correct++;
      fb.className = 'feedback-box correct';
      fb.textContent = '✓ ' + q.feedback;
    } else {
      fb.className = 'feedback-box incorrect';
      fb.textContent = '✗ ' + q.feedback;
      // Mark the correct answer
      const correctLbl = document.getElementById(`opt_${modIdx}_${qi}_${q.correct}`);
      if (correctLbl) correctLbl.classList.add('missed');
    }
  });

  if (!allAnswered) { alert('Please answer all questions before submitting.'); return; }

  const score = Math.round((correct / mod.questions.length) * 100);
  const passed = score >= 70;
  moduleScores[modIdx] = score;

  const scoreEl = document.getElementById(`scoreDisplay_${modIdx}`);
  scoreEl.textContent = `Score: ${correct}/${mod.questions.length} (${score}%)`;
  scoreEl.style.color = passed ? 'var(--green)' : 'var(--red)';

  const resultBox = document.getElementById(`resultBox_${modIdx}`);
  resultBox.style.display = 'block';
  if (passed) {
    resultBox.className = 'result-box pass';
    resultBox.innerHTML = `<h4>✅ Passed! (${score}%)</h4><p>Great work! You've completed this module. Continue to the next module.</p>`;
    completedModules.add(modIdx);
    document.getElementById(`nextBtn_${modIdx}`).disabled = false;
    const statusEl = document.getElementById(`sidebarStatus_${modIdx}`);
    if (statusEl) { statusEl.textContent = '✓'; document.getElementById(`sidebarBtn_${modIdx}`).classList.add('completed'); }
  } else {
    resultBox.className = 'result-box fail';
    resultBox.innerHTML = `<h4>❌ Not Passed (${score}%)</h4><p>You need 70% to pass. Review the content above and try again. You may retake as many times as needed.</p>`;
    document.getElementById(`retryBtn_${modIdx}`).style.display = 'inline-flex';
  }

  // Hide submit button
  document.querySelector(`#quiz_${modIdx} .btn-primary`).style.display = 'none';
  updateProgress();
}

function retryQuiz(modIdx) {
  const mod = modules[modIdx];
  mod.questions.forEach((q, qi) => {
    document.querySelectorAll(`input[name="q_${modIdx}_${qi}"]`).forEach(r => { r.disabled = false; r.checked = false; });
    q.options.forEach((opt, oi) => {
      const lbl = document.getElementById(`opt_${modIdx}_${qi}_${oi}`);
      lbl.className = 'option-label';
    });
    const fb = document.getElementById(`feedback_${modIdx}_${qi}`);
    fb.className = 'feedback-box';
    fb.textContent = '';
  });
  document.getElementById(`scoreDisplay_${modIdx}`).textContent = '';
  document.getElementById(`resultBox_${modIdx}`).style.display = 'none';
  document.getElementById(`retryBtn_${modIdx}`).style.display = 'none';
  document.querySelector(`#quiz_${modIdx} .btn-primary`).style.display = 'inline-flex';
}

function nextModule(modIdx) {
  if (modIdx < modules.length - 1) {
    showModule(modIdx + 1);
  } else {
    showCertificate();
  }
}

  cert.scrollIntoView({ behavior: 'smooth', block: 'start' });
  document.getElementById('certName').textContent = workerName;
  const today = new Date();
  document.getElementById('certDate').textContent = `Completed: ${today.toLocaleDateString('en-US', { year: 'numeric', month: 'long', day: 'numeric' })}`;
}

function updateProgress() {
  const total = modules.length;
  const done = completedModules.size;
  const pct = Math.round((done / total) * 100);
  document.getElementById('progressFill').style.width = pct + '%';
  document.getElementById('progressLabel').textContent = `${done} / ${total} Complete`;
}

function restartCourse() {
  completedModules = new Set();
  moduleScores = {};
  currentModule = -1;
  document.getElementById('certScreen').classList.remove('visible');
  document.getElementById('welcomeScreen').classList.remove('hidden');
  document.getElementById('moduleCards').innerHTML = '';
  document.getElementById('sidebarNav').innerHTML = '';
  updateProgress();
  window.scrollTo(0, 0);
}
